# CMake 进阶

## 1. 多目录项目

### 1.1 项目结构

```
myproject/
├── CMakeLists.txt          # 顶层 CMakeLists.txt
├── include/
│   └── mylib/
│       └── utils.h
├── src/
│   ├── CMakeLists.txt
│   ├── main.c
│   └── utils.c
├── lib/
│   ├── CMakeLists.txt
│   └── helper.c
└── tests/
    ├── CMakeLists.txt
    └── test_utils.c
```

### 1.2 顶层 CMakeLists.txt

```cmake
cmake_minimum_required(VERSION 3.10)
project(MyProject VERSION 1.0.0 LANGUAGES C)

# 全局设置
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# 选项
option(BUILD_TESTS "Build tests" ON)
option(BUILD_SHARED_LIBS "Build shared libraries" OFF)

# 添加子目录
add_subdirectory(lib)
add_subdirectory(src)

if(BUILD_TESTS)
    enable_testing()
    add_subdirectory(tests)
endif()
```

### 1.3 lib/CMakeLists.txt

```cmake
# 创建库
add_library(helper helper.c)

# 设置头文件路径（对使用者可见）
target_include_directories(helper PUBLIC
    $<BUILD_INTERFACE:${PROJECT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>
)
```

### 1.4 src/CMakeLists.txt

```cmake
# 创建可执行文件
add_executable(myapp main.c utils.c)

# 链接库
target_link_libraries(myapp PRIVATE helper)

# 头文件路径
target_include_directories(myapp PRIVATE
    ${PROJECT_SOURCE_DIR}/include
)
```

### 1.5 tests/CMakeLists.txt

```cmake
add_executable(test_utils test_utils.c)
target_link_libraries(test_utils PRIVATE helper)
target_include_directories(test_utils PRIVATE
    ${PROJECT_SOURCE_DIR}/include
)

# 添加测试
add_test(NAME test_utils COMMAND test_utils)
```

## 2. 查找和使用第三方库

### 2.1 find_package

`find_package` 用于查找系统中已安装的库：

```cmake
# 查找必需的库
find_package(Threads REQUIRED)
find_package(OpenSSL REQUIRED)

# 查找可选的库
find_package(ZLIB)
if(ZLIB_FOUND)
    target_link_libraries(app PRIVATE ZLIB::ZLIB)
endif()

# 使用找到的库
add_executable(app main.c)
target_link_libraries(app PRIVATE
    Threads::Threads
    OpenSSL::SSL
    OpenSSL::Crypto
)
```

### 2.2 常用库的查找

```cmake
# POSIX 线程
find_package(Threads REQUIRED)
target_link_libraries(app PRIVATE Threads::Threads)

# 数学库
target_link_libraries(app PRIVATE m)

# OpenSSL
find_package(OpenSSL REQUIRED)
target_link_libraries(app PRIVATE OpenSSL::SSL OpenSSL::Crypto)

# CURL
find_package(CURL REQUIRED)
target_link_libraries(app PRIVATE CURL::libcurl)

# SQLite3
find_package(SQLite3 REQUIRED)
target_link_libraries(app PRIVATE SQLite::SQLite3)

# Boost
find_package(Boost 1.70 REQUIRED COMPONENTS filesystem system)
target_link_libraries(app PRIVATE Boost::filesystem Boost::system)
```

### 2.3 pkg-config 支持

对于使用 pkg-config 的库：

```cmake
find_package(PkgConfig REQUIRED)

# 查找 libpng
pkg_check_modules(PNG REQUIRED libpng)
target_include_directories(app PRIVATE ${PNG_INCLUDE_DIRS})
target_link_libraries(app PRIVATE ${PNG_LIBRARIES})

# 更现代的方式（CMake 3.6+）
pkg_check_modules(PNG REQUIRED IMPORTED_TARGET libpng)
target_link_libraries(app PRIVATE PkgConfig::PNG)
```

### 2.4 手动指定库路径

当库没有 CMake 支持时：

```cmake
# 查找头文件
find_path(MYLIB_INCLUDE_DIR mylib.h
    PATHS /usr/local/include /opt/mylib/include
)

# 查找库文件
find_library(MYLIB_LIBRARY mylib
    PATHS /usr/local/lib /opt/mylib/lib
)

# 使用
if(MYLIB_INCLUDE_DIR AND MYLIB_LIBRARY)
    add_executable(app main.c)
    target_include_directories(app PRIVATE ${MYLIB_INCLUDE_DIR})
    target_link_libraries(app PRIVATE ${MYLIB_LIBRARY})
endif()
```

## 3. 创建和安装库

### 3.1 创建可安装的库

```cmake
cmake_minimum_required(VERSION 3.14)
project(MyLib VERSION 1.0.0 LANGUAGES C)

# 创建库
add_library(mylib
    src/mylib.c
    src/utils.c
)

# 设置版本
set_target_properties(mylib PROPERTIES
    VERSION ${PROJECT_VERSION}
    SOVERSION 1
)

# 头文件路径
target_include_directories(mylib PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>
)

# 安装库
install(TARGETS mylib
    EXPORT MyLibTargets
    LIBRARY DESTINATION lib
    ARCHIVE DESTINATION lib
    RUNTIME DESTINATION bin
)

# 安装头文件
install(DIRECTORY include/
    DESTINATION include
)

# 导出目标
install(EXPORT MyLibTargets
    FILE MyLibTargets.cmake
    NAMESPACE MyLib::
    DESTINATION lib/cmake/MyLib
)

# 生成配置文件
include(CMakePackageConfigHelpers)

configure_package_config_file(
    ${CMAKE_CURRENT_SOURCE_DIR}/cmake/MyLibConfig.cmake.in
    ${CMAKE_CURRENT_BINARY_DIR}/MyLibConfig.cmake
    INSTALL_DESTINATION lib/cmake/MyLib
)

write_basic_package_version_file(
    ${CMAKE_CURRENT_BINARY_DIR}/MyLibConfigVersion.cmake
    VERSION ${PROJECT_VERSION}
    COMPATIBILITY SameMajorVersion
)

install(FILES
    ${CMAKE_CURRENT_BINARY_DIR}/MyLibConfig.cmake
    ${CMAKE_CURRENT_BINARY_DIR}/MyLibConfigVersion.cmake
    DESTINATION lib/cmake/MyLib
)
```

### 3.2 cmake/MyLibConfig.cmake.in

```cmake
@PACKAGE_INIT@

include("${CMAKE_CURRENT_LIST_DIR}/MyLibTargets.cmake")

check_required_components(MyLib)
```

### 3.3 使用已安装的库

```cmake
find_package(MyLib 1.0 REQUIRED)
target_link_libraries(app PRIVATE MyLib::mylib)
```

## 4. FetchContent - 下载依赖

CMake 3.11+ 提供了 `FetchContent` 模块，可以在配置时下载依赖。

### 4.1 基本用法

```cmake
include(FetchContent)

# 声明依赖
FetchContent_Declare(
    json
    GIT_REPOSITORY https://github.com/nlohmann/json.git
    GIT_TAG v3.11.2
)

# 下载并添加到项目
FetchContent_MakeAvailable(json)

# 使用
target_link_libraries(app PRIVATE nlohmann_json::nlohmann_json)
```

### 4.2 多个依赖

```cmake
include(FetchContent)

FetchContent_Declare(
    googletest
    GIT_REPOSITORY https://github.com/google/googletest.git
    GIT_TAG v1.14.0
)

FetchContent_Declare(
    spdlog
    GIT_REPOSITORY https://github.com/gabime/spdlog.git
    GIT_TAG v1.12.0
)

# 一次性下载所有依赖
FetchContent_MakeAvailable(googletest spdlog)

# 使用
target_link_libraries(tests PRIVATE GTest::gtest_main)
target_link_libraries(app PRIVATE spdlog::spdlog)
```

### 4.3 下载源码包

```cmake
FetchContent_Declare(
    zlib
    URL https://zlib.net/zlib-1.3.tar.gz
    URL_HASH SHA256=ff0ba4c292013dbc27530b3a81e1f9a813cd39de01ca5e0f8bf355702efa593e
)

FetchContent_MakeAvailable(zlib)
```

## 5. 生成器表达式

生成器表达式在生成构建系统时求值，用于条件编译。

### 5.1 基本语法

```cmake
$<condition:value>           # 条件为真时返回 value
$<IF:condition,true,false>   # 条件判断
```

### 5.2 常用表达式

```cmake
# 配置类型判断
target_compile_definitions(app PRIVATE
    $<$<CONFIG:Debug>:DEBUG_MODE>
    $<$<CONFIG:Release>:NDEBUG>
)

# 编译器判断
target_compile_options(app PRIVATE
    $<$<CXX_COMPILER_ID:GNU>:-Wall -Wextra>
    $<$<CXX_COMPILER_ID:MSVC>:/W4>
)

# 平台判断
target_compile_definitions(app PRIVATE
    $<$<PLATFORM_ID:Linux>:LINUX_PLATFORM>
    $<$<PLATFORM_ID:Windows>:WINDOWS_PLATFORM>
)

# 目标属性
target_include_directories(mylib PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>
)
```

### 5.3 实用示例

```cmake
# 根据构建类型设置优化级别
target_compile_options(app PRIVATE
    $<$<CONFIG:Debug>:-O0 -g>
    $<$<CONFIG:Release>:-O3>
    $<$<CONFIG:RelWithDebInfo>:-O2 -g>
)

# 根据编译器添加特定选项
target_compile_options(app PRIVATE
    $<$<CXX_COMPILER_ID:GNU>:
        -Wall -Wextra -Wpedantic
        $<$<VERSION_GREATER_EQUAL:$<CXX_COMPILER_VERSION>,9.0>:-Wno-deprecated-copy>
    >
)
```

## 6. 自定义命令和目标

### 6.1 add_custom_command

在构建过程中执行自定义命令：

```cmake
# 生成头文件
add_custom_command(
    OUTPUT ${CMAKE_BINARY_DIR}/generated/config.h
    COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_BINARY_DIR}/generated
    COMMAND ${CMAKE_COMMAND} -E echo "#define VERSION \"${PROJECT_VERSION}\"" > ${CMAKE_BINARY_DIR}/generated/config.h
    COMMENT "Generating config.h"
)

# 将生成的文件添加到目标
add_executable(app main.c ${CMAKE_BINARY_DIR}/generated/config.h)
target_include_directories(app PRIVATE ${CMAKE_BINARY_DIR}/generated)
```

### 6.2 add_custom_target

创建自定义目标：

```cmake
# 格式化代码
add_custom_target(format
    COMMAND clang-format -i ${CMAKE_SOURCE_DIR}/src/*.c ${CMAKE_SOURCE_DIR}/include/*.h
    COMMENT "Formatting source code..."
    WORKING_DIRECTORY ${CMAKE_SOURCE_DIR}
)

# 生成文档
add_custom_target(docs
    COMMAND doxygen ${CMAKE_SOURCE_DIR}/Doxyfile
    COMMENT "Generating documentation..."
    WORKING_DIRECTORY ${CMAKE_SOURCE_DIR}
)

# 打包发布
add_custom_target(dist
    COMMAND ${CMAKE_COMMAND} -E tar czf ${PROJECT_NAME}-${PROJECT_VERSION}.tar.gz --format=gzip -- src include CMakeLists.txt
    COMMENT "Creating distribution package..."
    WORKING_DIRECTORY ${CMAKE_SOURCE_DIR}
)
```

### 6.3 目标依赖

```cmake
# 确保 docs 在 app 构建后执行
add_dependencies(docs app)

# 构建前生成文件
add_custom_target(generate_version ALL
    COMMAND ${CMAKE_COMMAND} -E echo "Building version ${PROJECT_VERSION}"
)
add_dependencies(app generate_version)
```

## 7. 测试支持

### 7.1 CTest 基础

```cmake
# 启用测试
enable_testing()

# 添加测试可执行文件
add_executable(test_math test_math.c)
target_link_libraries(test_math PRIVATE mylib)

# 注册测试
add_test(NAME test_math COMMAND test_math)

# 带参数的测试
add_test(NAME test_add COMMAND test_math --test=add)
add_test(NAME test_sub COMMAND test_math --test=sub)

# 设置测试属性
set_tests_properties(test_math PROPERTIES
    TIMEOUT 30
    LABELS "unit"
)
```

### 7.2 运行测试

```bash
# 构建后运行测试
cd build
ctest

# 详细输出
ctest -V

# 运行特定测试
ctest -R test_math

# 并行运行
ctest -j4

# 运行特定标签的测试
ctest -L unit
```

### 7.3 使用 Google Test

```cmake
include(FetchContent)

FetchContent_Declare(
    googletest
    GIT_REPOSITORY https://github.com/google/googletest.git
    GIT_TAG v1.14.0
)
FetchContent_MakeAvailable(googletest)

enable_testing()

add_executable(tests
    tests/test_main.cpp
    tests/test_utils.cpp
)
target_link_libraries(tests PRIVATE
    mylib
    GTest::gtest_main
)

include(GoogleTest)
gtest_discover_tests(tests)
```

## 8. 跨平台支持

### 8.1 平台检测

```cmake
if(WIN32)
    message(STATUS "Building for Windows")
    # Windows 特定设置
elseif(APPLE)
    message(STATUS "Building for macOS")
    # macOS 特定设置
elseif(UNIX)
    message(STATUS "Building for Linux/Unix")
    # Linux 特定设置
endif()
```

### 8.2 编译器检测

```cmake
if(CMAKE_C_COMPILER_ID STREQUAL "GNU")
    # GCC
    target_compile_options(app PRIVATE -Wall -Wextra)
elseif(CMAKE_C_COMPILER_ID STREQUAL "Clang")
    # Clang
    target_compile_options(app PRIVATE -Wall -Wextra)
elseif(CMAKE_C_COMPILER_ID STREQUAL "MSVC")
    # Visual Studio
    target_compile_options(app PRIVATE /W4)
endif()
```

### 8.3 功能检测

```cmake
include(CheckSymbolExists)
include(CheckIncludeFile)

# 检查函数是否存在
check_symbol_exists(strlcpy "string.h" HAVE_STRLCPY)
if(HAVE_STRLCPY)
    target_compile_definitions(app PRIVATE HAVE_STRLCPY)
endif()

# 检查头文件
check_include_file("sys/epoll.h" HAVE_EPOLL)
if(HAVE_EPOLL)
    target_compile_definitions(app PRIVATE HAVE_EPOLL)
endif()
```

## 9. 实用项目模板

### 9.1 完整的 C 项目模板

```cmake
# ============================================
#              项目配置
# ============================================
cmake_minimum_required(VERSION 3.14)

project(MyProject
    VERSION 1.0.0
    DESCRIPTION "My awesome project"
    LANGUAGES C
)

# ============================================
#              全局设置
# ============================================
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)

# 导出编译命令（用于 IDE 和 clangd）
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# 默认构建类型
if(NOT CMAKE_BUILD_TYPE AND NOT CMAKE_CONFIGURATION_TYPES)
    set(CMAKE_BUILD_TYPE Release CACHE STRING "Build type" FORCE)
    set_property(CACHE CMAKE_BUILD_TYPE PROPERTY STRINGS
        "Debug" "Release" "MinSizeRel" "RelWithDebInfo")
endif()

# ============================================
#              选项
# ============================================
option(BUILD_SHARED_LIBS "Build shared libraries" OFF)
option(BUILD_TESTS "Build tests" ON)
option(ENABLE_WARNINGS "Enable compiler warnings" ON)
option(ENABLE_SANITIZERS "Enable sanitizers in Debug" OFF)

# ============================================
#              编译选项
# ============================================
add_library(project_warnings INTERFACE)

if(ENABLE_WARNINGS)
    target_compile_options(project_warnings INTERFACE
        $<$<C_COMPILER_ID:GNU,Clang>:
            -Wall -Wextra -Wpedantic
            -Wshadow -Wcast-align -Wunused
            -Wconversion -Wsign-conversion
        >
        $<$<C_COMPILER_ID:MSVC>:/W4>
    )
endif()

# 调试时启用 Sanitizers
if(ENABLE_SANITIZERS AND CMAKE_BUILD_TYPE STREQUAL "Debug")
    add_compile_options(-fsanitize=address,undefined)
    add_link_options(-fsanitize=address,undefined)
endif()

# ============================================
#              依赖
# ============================================
find_package(Threads REQUIRED)

# ============================================
#              子目录
# ============================================
add_subdirectory(src)

if(BUILD_TESTS)
    enable_testing()
    add_subdirectory(tests)
endif()

# ============================================
#              安装
# ============================================
include(GNUInstallDirs)

install(TARGETS myapp
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)

install(DIRECTORY include/
    DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}
)

# ============================================
#              CPack 打包
# ============================================
set(CPACK_PACKAGE_NAME ${PROJECT_NAME})
set(CPACK_PACKAGE_VERSION ${PROJECT_VERSION})
set(CPACK_PACKAGE_VENDOR "MyCompany")
set(CPACK_PACKAGE_DESCRIPTION_SUMMARY ${PROJECT_DESCRIPTION})
include(CPack)

# ============================================
#              信息输出
# ============================================
message(STATUS "")
message(STATUS "=== ${PROJECT_NAME} v${PROJECT_VERSION} ===")
message(STATUS "Build type:       ${CMAKE_BUILD_TYPE}")
message(STATUS "C Compiler:       ${CMAKE_C_COMPILER_ID} ${CMAKE_C_COMPILER_VERSION}")
message(STATUS "Build tests:      ${BUILD_TESTS}")
message(STATUS "Shared libs:      ${BUILD_SHARED_LIBS}")
message(STATUS "Install prefix:   ${CMAKE_INSTALL_PREFIX}")
message(STATUS "")
```

### 9.2 常用 CMake 命令速查

```bash
# 配置
cmake -B build                           # 生成到 build 目录
cmake -B build -DCMAKE_BUILD_TYPE=Debug  # Debug 构建
cmake -B build -G Ninja                  # 使用 Ninja 生成器
cmake -B build -DBUILD_TESTS=OFF         # 禁用测试

# 构建
cmake --build build                      # 构建
cmake --build build -j4                  # 并行构建
cmake --build build --target clean       # 清理
cmake --build build --config Release     # 多配置生成器

# 安装
cmake --install build                    # 安装
cmake --install build --prefix /opt/app  # 指定安装目录

# 测试
ctest --test-dir build                   # 运行测试
ctest --test-dir build -V                # 详细输出

# 打包
cpack --config build/CPackConfig.cmake   # 生成安装包
```

## 10. 关键要点

1. **多目录项目**：使用 `add_subdirectory` 组织代码
2. **查找库**：`find_package` 查找系统库，`FetchContent` 下载依赖
3. **生成器表达式**：`$<...>` 实现条件编译
4. **现代 CMake**：优先使用 `target_*` 命令
5. **测试**：`enable_testing()` + `add_test()` 集成 CTest
6. **跨平台**：条件判断处理不同平台差异
7. **安装**：`install()` + `GNUInstallDirs` 标准化安装
