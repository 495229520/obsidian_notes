# 云计算案例：远程表达式计算器

## 概述

本案例实现一个**远程计算服务**：客户端将数学表达式发送给服务器，服务器计算后返回结果。这是一个简化的"云计算"模型，展示了如何通过网络进行远程过程调用（RPC）的基本思想。

```
客户端                                服务器
   │                                    │
   │  发送: "3 + 5"                      │
   │ ───────────────────────────────────→│
   │                                    │ 解析表达式
   │                                    │ 计算: 3 + 5 = 8
   │  返回: "8"                          │
   │←───────────────────────────────────│
```

### 应用场景

| 场景 | 说明 |
|:---|:---|
| 云计算 | 将计算密集型任务交给服务器处理 |
| 分布式计算 | 多个客户端共享服务器的计算资源 |
| 瘦客户端 | 客户端只负责输入输出，计算在服务器完成 |

## 数据结构设计

为了在客户端和服务器之间传输结构化数据，我们定义一个**通信协议结构体**：

```c
// 计算请求结构体
typedef struct {
    int operand1;      // 第一个操作数
    int operand2;      // 第二个操作数
    char operator;     // 运算符 (+, -, *, /)
    char padding[3];   // 填充字节，保证结构体对齐
} CalcRequest;

// 计算结果结构体
typedef struct {
    int result;        // 计算结果
    int error;         // 错误码 (0 表示成功)
} CalcResponse;
```

### 结构体内存布局

```
CalcRequest 内存布局 (8 字节):
┌──────────┬──────────┬─────────┬─────────┐
│ operand1 │ operand2 │ operator│ padding │
│  4 bytes │  4 bytes │ 1 byte  │ 3 bytes │
└──────────┴──────────┴─────────┴─────────┘

CalcResponse 内存布局 (8 字节):
┌──────────┬──────────┐
│  result  │  error   │
│  4 bytes │  4 bytes │
└──────────┴──────────┘
```

> **注意**：在实际网络编程中，需要考虑字节序问题。跨平台通信时应使用 `htonl()`/`ntohl()` 进行字节序转换。

## 服务端实现

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>

// 数据结构定义
typedef struct {
    int operand1;
    int operand2;
    char op;
    char padding[3];
} CalcRequest;

typedef struct {
    int result;
    int error;
} CalcResponse;

// 计算函数
int calculate(int a, int b, char op, int *error) {
    *error = 0;
    switch (op) {
        case '+': return a + b;
        case '-': return a - b;
        case '*': return a * b;
        case '/':
            if (b == 0) {
                *error = 1;  // 除零错误
                return 0;
            }
            return a / b;
        default:
            *error = 2;  // 不支持的运算符
            return 0;
    }
}

void run_server() {
    // 1. 创建套接字
    int listen_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (listen_fd < 0) {
        perror("socket");
        return;
    }

    // 设置地址复用
    int opt = 1;
    setsockopt(listen_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    // 2. 绑定地址
    struct sockaddr_in serveraddr;
    memset(&serveraddr, 0, sizeof(serveraddr));
    serveraddr.sin_family = AF_INET;
    serveraddr.sin_port = htons(9527);
    serveraddr.sin_addr.s_addr = htonl(INADDR_ANY);

    if (bind(listen_fd, (struct sockaddr*)&serveraddr, sizeof(serveraddr)) < 0) {
        perror("bind");
        close(listen_fd);
        return;
    }

    // 3. 开始监听
    if (listen(listen_fd, 5) < 0) {
        perror("listen");
        close(listen_fd);
        return;
    }

    printf("Calculator server started on port 9527...\n");

    // 4. 主循环：接受连接并处理计算请求
    while (1) {
        struct sockaddr_in clientaddr;
        socklen_t clientlen = sizeof(clientaddr);

        int conn_fd = accept(listen_fd, (struct sockaddr*)&clientaddr, &clientlen);
        if (conn_fd < 0) {
            perror("accept");
            continue;
        }

        printf("Client connected: %s:%d\n",
               inet_ntoa(clientaddr.sin_addr),
               ntohs(clientaddr.sin_port));

        // 5. 接收计算请求
        CalcRequest request;
        CalcResponse response;

        // 循环读取，确保收到完整的请求数据
        int total_read = 0;
        while (total_read < sizeof(CalcRequest)) {
            int len = recv(conn_fd, (char*)&request + total_read,
                          sizeof(CalcRequest) - total_read, 0);
            if (len <= 0) {
                printf("Client disconnected\n");
                break;
            }
            total_read += len;
        }

        if (total_read == sizeof(CalcRequest)) {
            printf("Received: %d %c %d\n",
                   request.operand1, request.op, request.operand2);

            // 6. 执行计算
            response.result = calculate(request.operand1,
                                        request.operand2,
                                        request.op,
                                        &response.error);

            // 7. 发送结果
            send(conn_fd, &response, sizeof(CalcResponse), 0);

            if (response.error == 0) {
                printf("Result: %d\n", response.result);
            } else {
                printf("Error: %d\n", response.error);
            }
        }

        close(conn_fd);
    }

    close(listen_fd);
}

int main() {
    run_server();
    return 0;
}
```

## 客户端实现

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>

typedef struct {
    int operand1;
    int operand2;
    char op;
    char padding[3];
} CalcRequest;

typedef struct {
    int result;
    int error;
} CalcResponse;

int send_calculation(int a, char op, int b) {
    // 1. 创建套接字
    int client_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (client_fd < 0) {
        perror("socket");
        return -1;
    }

    // 2. 连接服务器
    struct sockaddr_in serveraddr;
    memset(&serveraddr, 0, sizeof(serveraddr));
    serveraddr.sin_family = AF_INET;
    serveraddr.sin_port = htons(9527);
    serveraddr.sin_addr.s_addr = inet_addr("127.0.0.1");

    if (connect(client_fd, (struct sockaddr*)&serveraddr, sizeof(serveraddr)) < 0) {
        perror("connect");
        close(client_fd);
        return -1;
    }

    // 3. 构造请求
    CalcRequest request;
    request.operand1 = a;
    request.operand2 = b;
    request.op = op;
    memset(request.padding, 0, sizeof(request.padding));

    // 4. 发送请求
    send(client_fd, &request, sizeof(CalcRequest), 0);
    printf("Sent: %d %c %d\n", a, op, b);

    // 5. 接收结果（循环读取确保完整）
    CalcResponse response;
    int total_read = 0;
    while (total_read < sizeof(CalcResponse)) {
        int len = recv(client_fd, (char*)&response + total_read,
                      sizeof(CalcResponse) - total_read, 0);
        if (len <= 0) {
            printf("Server disconnected\n");
            close(client_fd);
            return -1;
        }
        total_read += len;
    }

    // 6. 显示结果
    if (response.error == 0) {
        printf("Result: %d %c %d = %d\n", a, op, b, response.result);
    } else {
        printf("Calculation error: %d\n", response.error);
    }

    close(client_fd);
    return response.result;
}

int main() {
    // 测试各种计算
    send_calculation(10, '+', 5);   // 15
    send_calculation(10, '-', 3);   // 7
    send_calculation(6, '*', 7);    // 42
    send_calculation(20, '/', 4);   // 5
    send_calculation(10, '/', 0);   // 错误：除零

    return 0;
}
```

## 代码结构分析

### 第一部分：协议定义

```c
// 请求和响应结构体定义了客户端与服务器之间的通信协议
typedef struct {
    int operand1;      // 操作数1
    int operand2;      // 操作数2
    char op;           // 运算符
    char padding[3];   // 填充对齐
} CalcRequest;
```

**关键点**：
- 使用固定大小的结构体确保传输数据的可预测性
- `padding` 用于保证结构体大小是 4 的倍数（[[1.3 进程的虚拟地址空间|内存对齐]]）

### 第二部分：服务器网络处理

服务器遵循标准的 [[7.7 listen和accept函数|TCP 服务器流程]]：

```
socket() → bind() → listen() → accept() → recv() → 处理 → send() → close()
```

### 第三部分：计算逻辑

```c
int calculate(int a, int b, char op, int *error) {
    switch (op) {
        case '+': return a + b;
        case '-': return a - b;
        case '*': return a * b;
        case '/':
            if (b == 0) {
                *error = 1;
                return 0;
            }
            return a / b;
        default:
            *error = 2;
            return 0;
    }
}
```

## 详细执行流程

### 1. 服务器启动

```
服务器
   │
   ├── socket() ──→ 创建 TCP [[7.6 套接字|套接字]]
   │
   ├── setsockopt() ──→ 设置地址复用选项
   │
   ├── bind() ──→ 绑定到 0.0.0.0:9527
   │
   ├── listen() ──→ 开始监听，等待队列长度 5
   │
   └── accept() ──→ [阻塞等待客户端...]
```

### 2. 客户端发起计算请求

```
客户端                                      服务器
   │                                          │
   │  socket()                                │
   │                                          │
   │  connect(127.0.0.1:9527)                 │
   │ ─────────────── [[7.5 TCP协议基础#TCP 三次握手（建立连接）|三次握手]] ───────────→│
   │                                          │ accept() 返回
   │                                          │
   │  构造 CalcRequest: {10, 5, '+'}          │
   │  send(request)                           │
   │ ────────────────────────────────────────→│ recv() 收到请求
   │                                          │
   │                                          │ calculate(10, 5, '+')
   │                                          │ result = 15
   │                                          │
   │  recv() 等待结果                          │ send(response)
   │←────────────────────────────────────────│
   │                                          │
   │  打印: "10 + 5 = 15"                      │
   │                                          │
   │  close()                                 │ close()
```

### 3. 数据传输细节

```
发送 CalcRequest (8字节):
┌────────────────────────────────────────┐
│ 00 00 00 0A │ 00 00 00 05 │ 2B │ 00 00 00 │
│  operand1   │  operand2   │ op │ padding  │
│   (10)      │    (5)      │('+')│         │
└────────────────────────────────────────┘

接收 CalcResponse (8字节):
┌────────────────────────────────────────┐
│ 00 00 00 0F │ 00 00 00 00 │
│   result    │    error    │
│    (15)     │    (0)      │
└────────────────────────────────────────┘
```

## 循环读取的重要性

如同 [[8.5 回声服务器#TCP 字节流特性与数据接收问题|回声服务器中所讨论的]]，由于 TCP 是字节流协议，必须循环读取确保收到完整数据：

```c
// 接收完整的 CalcRequest
int total_read = 0;
while (total_read < sizeof(CalcRequest)) {
    int len = recv(conn_fd, (char*)&request + total_read,
                  sizeof(CalcRequest) - total_read, 0);
    if (len <= 0) break;
    total_read += len;
}
```

### 为什么需要强制类型转换？

```c
recv(conn_fd, (char*)&request + total_read, ...)
//            ^^^^^^
```

- `&request` 是 `CalcRequest*` 类型
- 指针运算以元素大小为单位：`request + 1` 会偏移 `sizeof(CalcRequest)` 字节
- 转换为 `(char*)` 后，`+total_read` 只偏移 `total_read` 字节
- 这样才能实现"从当前已读位置继续读取"的效果

## 扩展：使用 [[5.1 系统进程控制函数#1. fork() - 创建子进程|fork()]] 实现并发版本

```c
while (1) {
    int conn_fd = accept(listen_fd, ...);

    pid_t pid = fork();
    if (pid == 0) {
        // 子进程处理计算请求
        close(listen_fd);

        CalcRequest request;
        CalcResponse response;

        // 接收、计算、返回结果
        recv(conn_fd, &request, sizeof(request), 0);
        response.result = calculate(...);
        send(conn_fd, &response, sizeof(response), 0);

        close(conn_fd);
        exit(0);
    } else {
        // 父进程继续监听
        close(conn_fd);
    }
}
```

## 关键要点

1. **协议设计**：
   - 使用固定大小的结构体作为通信协议
   - 注意内存对齐和字节序问题
   - 定义错误码便于处理异常情况

2. **数据完整性**：
   - TCP 是字节流协议，必须循环读取
   - 发送和接收的数据量必须匹配协议定义

3. **与其他案例的关系**：
   - 基于 [[8.4 迭代服务器]] 的架构
   - 采用 [[8.5 回声服务器#TCP 字节流特性与数据接收问题|循环读取]]的数据接收方式
   - 业务逻辑替换为表达式计算

4. **实际应用启示**：
   - 这是 RPC（远程过程调用）的简化模型
   - 实际系统中会使用 Protocol Buffers、gRPC 等框架
   - 需要考虑安全性、超时、重试等问题
