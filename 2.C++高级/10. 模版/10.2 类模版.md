# 10.2 类模板

## 知识点

1. **类模板实例化后会变成模板类**
2. **模板名称 + 类型参数列表 = 类名称**
3. **构造和析构函数名可以不用加 `<T>`**，其他出现模版的地方都要需加上类型参数列表
4. **类模版有选择性实例化的特征**：只有调用了类模版的具体实现方法，类模版才会去实例化具体方法
5. 在类外实现类的成员方法需要带 `template` 并声明模版类的作用域

---

## 类模板基础

### 模板类成员方法的声明与定义

```cpp
// 类模版 => 实例化 => 模板类
// 添加默认的类型参数
template<typename T=int>
class SeqStack  // 模版名称+类型参数列表 = 类名称
{
public:
    // 构造和析构函数名可以不用加<T>，其他出现模版的地方都要需加上类型参数列表
    SeqStack<T>(int size = 10)
        :_pstack(new T[size])
        , _top(0)
        , _size(size)
    {}

    ~SeqStack<T>()
    {
        delete[] _pstack;
        _pstack = nullptr;
    }

    SeqStack<T>& operator=(const SeqStack<T>& stack)
    {
        if (this == &stack)
            return *this;

        delete[] _pstack;

        _top = stack._top;
        _size = stack._size;
        _pstack = new T[_size];
        for (int i = 0; i < _top; ++i)
        {
            _pstack[i] = stack._pstack[i];
        }
        return *this;
    }

    // 入栈操作
    void push(const T& val);

    // 出栈操作
    void pop()
    {
        if (empty())
            return;
        --_top;
    }

    T top() const    // 返回栈顶元素
    {
        if (empty())
            throw "stack is empty!";  // 抛出异常代表函数逻辑结束
        return _pstack[_top - 1];
    }

    bool empty() const { return _top == 0; }    // 栈空

private:
    T* _pstack;
    int _top;
    int _size;

    // 顺序栈底层数组按2倍的方式进行扩容
    void expand()
    {
        T* ptmp = new T[_size * 2];
        for (int i = 0; i < _top; i++)
        {
            ptmp[i] = _pstack[i];
        }
        delete[] _pstack;
        _pstack = ptmp;
        _size *= 2;
    }
};
```

### 在类外实现入栈操作

```cpp
// 在类外实现类的成员方法需要带template并声明模版类的作用域
template<typename T>
void SeqStack<T>::push(const T& val)
{
    if (full())
        expand();
    _pstack[_top++] = val;
}
```

## 类模版有选择性实例化的特征

### 示例

```cpp
template<typename T=int>
class SeqStack
{
    // ... 类定义
};

int main()
{
    // 类模板有选择性实例化的特征
    // 模版实例化生成模版类  class SeqStack<int>{};
    SeqStack<int> s1;
    s1.push(1);
    s1.push(2);
    s1.push(3);
    s1.push(4);
    s1.pop();
    cout << s1.top() << endl;
    return 0;
}
```

**输出**：

```
Test()
Test()
Test()
Test()
Test()
Test()
Test()
```

**说明**：只有调用了类模版的具体实现方法，类模版才会去实例化具体方法。例如，在 `main` 中没有看到 `Test()` 析构函数被调用。

## 完整案例：SeqStack 类模板

```cpp
#include <iostream>
using namespace std;

template<typename T=int>
class SeqStack
{
public:
    SeqStack<T>(int size = 10)
        :_pstack(new T[size])
        , _top(0)
        , _size(size)
    {}

    ~SeqStack<T>()
    {
        delete[] _pstack;
        _pstack = nullptr;
    }

    SeqStack<T>(const SeqStack<T>& stack)
        :_top(stack._top)
        , _size(stack._size)
    {
        _pstack = new T[_size];
        for (int i = 0; i < _top; ++i)
        {
            _pstack[i] = stack._pstack[i];
        }
    }

    SeqStack<T>& operator=(const SeqStack<T>& stack)
    {
        if (this == &stack)
            return *this;

        delete[] _pstack;

        _top = stack._top;
        _size = stack._size;
        _pstack = new T[_size];
        for (int i = 0; i < _top; ++i)
        {
            _pstack[i] = stack._pstack[i];
        }
        return *this;
    }

    void push(const T& val)
    {
        if (full())
            expand();
        _pstack[_top++] = val;
    }

    void pop()
    {
        if (empty())
            return;
        --_top;
    }

    T top() const
    {
        if (empty())
            throw "stack is empty!";
        return _pstack[_top - 1];
    }

    bool empty() const { return _top == 0; }
    bool full() const { return _top == _size; }

private:
    T* _pstack;
    int _top;
    int _size;

    void expand()
    {
        T* ptmp = new T[_size * 2];
        for (int i = 0; i < _top; i++)
        {
            ptmp[i] = _pstack[i];
        }
        delete[] _pstack;
        _pstack = ptmp;
        _size *= 2;
    }
};

int main()
{
    SeqStack<int> s1;
    s1.push(1);
    s1.push(2);
    s1.push(3);
    s1.push(4);
    s1.pop();
    cout << s1.top() << endl;  // 输出: 3
    return 0;
}
```

## 空间配置器

空间配置器（Allocator）是 STL 中用于管理内存分配和释放的组件。在类模板中使用空间配置器可以将**对象的内存分配**和**对象的构造**分离开来。

### 基本概念

```cpp
template<typename T, typename Alloc = allocator<T>>
class vector
{
    // ...
};
```

### 空间配置器的作用

1. **分离内存分配与对象构造**：`allocate()` 只分配内存，`construct()` 负责在已分配的内存上构造对象
2. **分离对象析构与内存释放**：`destroy()` 只析构对象，`deallocate()` 负责释放内存
3. **提高效率**：避免不必要的构造和析构操作

## vector 简单实现

> 完整的迭代器实现请参见 [[12.2 vector容器的迭代器实现]]

```cpp
template<typename T>
class vector
{
public:
    vector(int size = 10)
    {
        _first = new T[size];
        _last = _first;
        _end = _first + size;
    }

    ~vector()
    {
        delete[] _first;
        _first = _last = _end = nullptr;
    }

    vector(const vector<T>& rhs)
    {
        int size = rhs._end - rhs._first;
        _first = new T[size];
        int len = rhs._last - rhs._first;
        for (int i = 0; i < len; ++i)
        {
            _first[i] = rhs._first[i];
        }
        _last = _first + len;
        _end = _first + size;
    }

    vector<T>& operator=(const vector<T>& rhs)
    {
        if (this == &rhs)
            return *this;
        delete[] _first;
        int size = rhs._end - rhs._first;
        _first = new T[size];
        int len = rhs._last - rhs._first;
        for (int i = 0; i < len; ++i)
        {
            _first[i] = rhs._first[i];
        }
        _last = _first + len;
        _end = _first + size;
        return *this;
    }

    void push_back(const T& val)
    {
        if (full())
            expand();
        *_last++ = val;
    }

    void pop_back()
    {
        if (empty())
            return;
        --_last;
    }

    T back() const
    {
        return *(_last - 1);
    }

    bool full() const { return _last == _end; }
    bool empty() const { return _first == _last; }
    int size() const { return _last - _first; }

private:
    T* _first;    // 指向数组起始的位置
    T* _last;     // 指向数组中有效元素的后继位置
    T* _end;      // 指向数组空间的后继位置

    void expand()
    {
        int size = _end - _first;
        T* ptmp = new T[size * 2];
        for (int i = 0; i < size; ++i)
        {
            ptmp[i] = _first[i];
        }
        delete[] _first;
        _first = ptmp;
        _last = _first + size;
        _end = _first + size * 2;
    }
};

int main()
{
    vector<int> vec;
    for (int i = 0; i < 20; ++i)
    {
        vec.push_back(rand() % 100);
    }
    while (!vec.empty())
    {
        cout << vec.back() << " ";
        vec.pop_back();
    }
    return 0;
}
```

## 关键要点

1. **类模板 vs 模板类**：类模板实例化后才成为模板类
2. **类名表示**：模板名称 + 类型参数列表 = 完整类名
3. **选择性实例化**：只有被调用的成员方法才会实例化
4. **类外定义**：需要带 `template` 关键字和类型参数
5. **空间配置器**：分离内存管理和对象生命周期管理
