# 排行榜

> Redis 的有序集合（ZSet）天然支持按分数排序，是实现排行榜的最佳数据结构，可轻松应对百万级数据的实时排名查询。

---

## 概念定义

**排行榜** 是一种按特定指标对用户/对象进行排序展示的功能。Redis ZSet 的特点使其非常适合：

- **有序存储**：元素按 score 自动排序
- **O(log N) 复杂度**：增删改查都很快
- **丰富的排名操作**：支持正序、逆序、范围查询

---

## ZSet 基础命令

```bash
# 添加/更新元素（score, member）
ZADD leaderboard 9500 "player_A"
ZADD leaderboard 8800 "player_B" 9200 "player_C"

# 增加分数
ZINCRBY leaderboard 100 "player_A"

# 获取排名（从 0 开始，降序）
ZREVRANK leaderboard "player_A"

# 获取分数
ZSCORE leaderboard "player_A"

# 获取排名前 10（降序，带分数）
ZREVRANGE leaderboard 0 9 WITHSCORES

# 获取总人数
ZCARD leaderboard

# 删除元素
ZREM leaderboard "player_A"
```

## 基础排行榜实现

### 架构设计

```
┌─────────────────────────────────────────────────────────┐
│                     排行榜系统                           │
├─────────────────────────────────────────────────────────┤
│                                                         │
│  用户操作 ──► 业务服务 ──► Redis ZSet                   │
│                │                                        │
│                ▼                                        │
│           分数更新 ──► ZINCRBY leaderboard score user   │
│           查询排名 ──► ZREVRANK leaderboard user        │
│           排行榜   ──► ZREVRANGE leaderboard 0 N        │
│                                                         │
└─────────────────────────────────────────────────────────┘
```

### C++ 实现

```cpp
#include <hiredis/hiredis.h>
#include <string>
#include <vector>
#include <optional>

struct RankEntry {
    std::string member;
    double score;
    int64_t rank;  // 从 1 开始
};

class Leaderboard {
public:
    Leaderboard(redisContext* ctx, const std::string& name)
        : ctx_(ctx), key_("leaderboard:" + name)
    {}

    // 更新分数（覆盖）
    bool setScore(const std::string& member, double score) {
        auto* reply = static_cast<redisReply*>(
            redisCommand(ctx_, "ZADD %s %f %s",
                key_.c_str(), score, member.c_str())
        );

        bool success = reply && reply->type == REDIS_REPLY_INTEGER;
        if (reply) freeReplyObject(reply);
        return success;
    }

    // 增加分数
    double incrScore(const std::string& member, double increment) {
        auto* reply = static_cast<redisReply*>(
            redisCommand(ctx_, "ZINCRBY %s %f %s",
                key_.c_str(), increment, member.c_str())
        );

        double new_score = 0;
        if (reply && reply->type == REDIS_REPLY_STRING) {
            new_score = std::stod(reply->str);
        }

        if (reply) freeReplyObject(reply);
        return new_score;
    }

    // 获取排名（降序，从 1 开始）
    std::optional<int64_t> getRank(const std::string& member) {
        auto* reply = static_cast<redisReply*>(
            redisCommand(ctx_, "ZREVRANK %s %s",
                key_.c_str(), member.c_str())
        );

        std::optional<int64_t> rank;
        if (reply && reply->type == REDIS_REPLY_INTEGER) {
            rank = reply->integer + 1;  // Redis 从 0 开始，转为从 1 开始
        }

        if (reply) freeReplyObject(reply);
        return rank;
    }

    // 获取分数
    std::optional<double> getScore(const std::string& member) {
        auto* reply = static_cast<redisReply*>(
            redisCommand(ctx_, "ZSCORE %s %s",
                key_.c_str(), member.c_str())
        );

        std::optional<double> score;
        if (reply && reply->type == REDIS_REPLY_STRING) {
            score = std::stod(reply->str);
        }

        if (reply) freeReplyObject(reply);
        return score;
    }

    // 获取排行榜（前 N 名）
    std::vector<RankEntry> getTopN(int n) {
        auto* reply = static_cast<redisReply*>(
            redisCommand(ctx_, "ZREVRANGE %s 0 %d WITHSCORES",
                key_.c_str(), n - 1)
        );

        std::vector<RankEntry> result;

        if (reply && reply->type == REDIS_REPLY_ARRAY) {
            for (size_t i = 0; i < reply->elements; i += 2) {
                RankEntry entry;
                entry.member = std::string(reply->element[i]->str,
                                           reply->element[i]->len);
                entry.score = std::stod(reply->element[i + 1]->str);
                entry.rank = static_cast<int64_t>(i / 2 + 1);
                result.push_back(entry);
            }
        }

        if (reply) freeReplyObject(reply);
        return result;
    }

    // 获取指定排名范围
    std::vector<RankEntry> getRange(int start_rank, int end_rank) {
        auto* reply = static_cast<redisReply*>(
            redisCommand(ctx_, "ZREVRANGE %s %d %d WITHSCORES",
                key_.c_str(), start_rank - 1, end_rank - 1)
        );

        std::vector<RankEntry> result;

        if (reply && reply->type == REDIS_REPLY_ARRAY) {
            for (size_t i = 0; i < reply->elements; i += 2) {
                RankEntry entry;
                entry.member = std::string(reply->element[i]->str,
                                           reply->element[i]->len);
                entry.score = std::stod(reply->element[i + 1]->str);
                entry.rank = start_rank + static_cast<int64_t>(i / 2);
                result.push_back(entry);
            }
        }

        if (reply) freeReplyObject(reply);
        return result;
    }

    // 获取用户周围的排名（上下各 N 人）
    std::vector<RankEntry> getAroundUser(const std::string& member, int count) {
        auto rank = getRank(member);
        if (!rank) {
            return {};
        }

        int start = std::max(1, static_cast<int>(*rank) - count);
        int end = static_cast<int>(*rank) + count;

        return getRange(start, end);
    }

    // 获取总人数
    int64_t size() {
        auto* reply = static_cast<redisReply*>(
            redisCommand(ctx_, "ZCARD %s", key_.c_str())
        );

        int64_t count = 0;
        if (reply && reply->type == REDIS_REPLY_INTEGER) {
            count = reply->integer;
        }

        if (reply) freeReplyObject(reply);
        return count;
    }

    // 删除用户
    bool remove(const std::string& member) {
        auto* reply = static_cast<redisReply*>(
            redisCommand(ctx_, "ZREM %s %s",
                key_.c_str(), member.c_str())
        );

        bool success = reply && reply->type == REDIS_REPLY_INTEGER
                       && reply->integer > 0;
        if (reply) freeReplyObject(reply);
        return success;
    }

private:
    redisContext* ctx_;
    std::string key_;
};
```

**设计要点**：
- 排名转换：Redis 返回 0-based，业务层转为 1-based
- `WITHSCORES` 选项同时返回分数和成员
- `getAroundUser` 查询用户附近排名，常用于"我的排名"页面

### 使用示例

```cpp
void leaderboardDemo(redisContext* ctx) {
    Leaderboard board(ctx, "game:daily");

    // 添加玩家分数
    board.setScore("player_A", 9500);
    board.setScore("player_B", 8800);
    board.setScore("player_C", 9200);
    board.setScore("player_D", 8500);
    board.setScore("player_E", 9100);

    // 玩家 B 打了一局，加 500 分
    double new_score = board.incrScore("player_B", 500);
    std::cout << "player_B 新分数: " << new_score << '\n';

    // 查询 player_C 的排名
    if (auto rank = board.getRank("player_C")) {
        std::cout << "player_C 排名: " << *rank << '\n';
    }

    // 获取排行榜前 3 名
    std::cout << "\n=== 排行榜 Top 3 ===\n";
    for (const auto& entry : board.getTopN(3)) {
        std::cout << "第" << entry.rank << "名: "
                  << entry.member << " - " << entry.score << "分\n";
    }

    // 获取 player_D 周围的排名
    std::cout << "\n=== player_D 附近排名 ===\n";
    for (const auto& entry : board.getAroundUser("player_D", 2)) {
        std::cout << "第" << entry.rank << "名: "
                  << entry.member << " - " << entry.score << "分\n";
    }
}
```

**输出**：
```
player_B 新分数: 9300
player_C 排名: 3

=== 排行榜 Top 3 ===
第1名: player_A - 9500分
第2名: player_B - 9300分
第3名: player_C - 9200分

=== player_D 附近排名 ===
第3名: player_C - 9200分
第4名: player_E - 9100分
第5名: player_D - 8500分
```

## 分数相同时的排序问题

### 问题

ZSet 中分数相同的成员按字典序排列，这可能不符合业务需求（如：同分时先达到者排前面）。

```bash
ZADD board 1000 "user_A"  # 10:00 到达
ZADD board 1000 "user_B"  # 10:05 到达

ZREVRANGE board 0 1
# 返回：user_B, user_A（字典序，不是期望的先到先得）
```

### 解决方案：复合分数

将时间戳编码到分数的小数部分。

```cpp
class TimestampLeaderboard {
public:
    // 计算复合分数：整数部分是分数，小数部分是时间（越早越大）
    double composeScore(int64_t score, int64_t timestamp) {
        // 用一个大数减去时间戳，使早的时间得到更大的小数部分
        constexpr int64_t MAX_TIME = 9999999999999LL;  // 13 位时间戳上限
        double time_factor = static_cast<double>(MAX_TIME - timestamp) / MAX_TIME;
        return static_cast<double>(score) + time_factor * 0.999999;
    }

    // 从复合分数提取原始分数
    int64_t extractScore(double composed_score) {
        return static_cast<int64_t>(composed_score);
    }

    bool setScore(const std::string& member, int64_t score) {
        auto now = std::chrono::duration_cast<std::chrono::milliseconds>(
            std::chrono::system_clock::now().time_since_epoch()
        ).count();

        double composed = composeScore(score, now);

        auto* reply = static_cast<redisReply*>(
            redisCommand(ctx_, "ZADD %s %f %s",
                key_.c_str(), composed, member.c_str())
        );

        bool success = reply && reply->type == REDIS_REPLY_INTEGER;
        if (reply) freeReplyObject(reply);
        return success;
    }

    // ... 其他方法类似，获取分数时用 extractScore 转换

private:
    redisContext* ctx_;
    std::string key_;
};
```

**原理**：
- 分数 1000，时间 1703001600000 → 复合分数 1000.xxxxx
- 同分数时，时间戳小的 `MAX_TIME - timestamp` 更大，排名更靠前

## 周期性排行榜

### 场景

- **日榜**：每天 0 点重置
- **周榜**：每周一 0 点重置
- **月榜**：每月 1 号重置

### 实现方案

#### 方案一：Key 中编码时间

```cpp
class PeriodicLeaderboard {
public:
    enum class Period { Daily, Weekly, Monthly };

    PeriodicLeaderboard(redisContext* ctx, const std::string& name, Period period)
        : ctx_(ctx), base_key_("leaderboard:" + name), period_(period)
    {}

    std::string getCurrentKey() {
        return base_key_ + ":" + getPeriodSuffix();
    }

    bool setScore(const std::string& member, double score) {
        std::string key = getCurrentKey();
        auto* reply = static_cast<redisReply*>(
            redisCommand(ctx_, "ZADD %s %f %s",
                key.c_str(), score, member.c_str())
        );

        // 设置过期时间，自动清理历史数据
        setExpire(key);

        bool success = reply && reply->type == REDIS_REPLY_INTEGER;
        if (reply) freeReplyObject(reply);
        return success;
    }

private:
    redisContext* ctx_;
    std::string base_key_;
    Period period_;

    std::string getPeriodSuffix() {
        auto now = std::chrono::system_clock::now();
        std::time_t time = std::chrono::system_clock::to_time_t(now);
        std::tm* tm = std::localtime(&time);

        char buf[32];
        switch (period_) {
            case Period::Daily:
                std::strftime(buf, sizeof(buf), "%Y%m%d", tm);
                break;
            case Period::Weekly:
                std::strftime(buf, sizeof(buf), "%Y-W%W", tm);
                break;
            case Period::Monthly:
                std::strftime(buf, sizeof(buf), "%Y%m", tm);
                break;
        }
        return buf;
    }

    void setExpire(const std::string& key) {
        int ttl;
        switch (period_) {
            case Period::Daily:   ttl = 2 * 24 * 3600; break;   // 保留 2 天
            case Period::Weekly:  ttl = 14 * 24 * 3600; break;  // 保留 2 周
            case Period::Monthly: ttl = 60 * 24 * 3600; break;  // 保留 2 月
        }

        redisCommand(ctx_, "EXPIRE %s %d", key.c_str(), ttl);
    }
};
```

**优点**：
- 自动按时间隔离数据
- 过期自动清理
- 可以查询历史榜单

#### 方案二：定时任务重置

```bash
# 每天 0 点执行
DEL leaderboard:daily
# 或者重命名归档
RENAME leaderboard:daily leaderboard:daily:20231224
```

## 多维度排行榜

### 场景

游戏中可能有多个排行榜维度：

- 积分榜
- 等级榜
- 战力榜
- 充值榜

### 实现

```cpp
class MultiDimensionLeaderboard {
public:
    MultiDimensionLeaderboard(redisContext* ctx, const std::string& game)
        : ctx_(ctx), prefix_("rank:" + game + ":")
    {}

    void updateScore(const std::string& user,
                     const std::string& dimension,
                     double score) {
        std::string key = prefix_ + dimension;
        redisCommand(ctx_, "ZADD %s %f %s",
            key.c_str(), score, user.c_str());
    }

    // 批量更新多个维度
    void updateMultiple(const std::string& user,
                        const std::vector<std::pair<std::string, double>>& scores) {
        for (const auto& [dimension, score] : scores) {
            updateScore(user, dimension, score);
        }
    }

    // 获取用户在所有维度的排名
    std::map<std::string, int64_t> getAllRanks(const std::string& user,
                                                const std::vector<std::string>& dimensions) {
        std::map<std::string, int64_t> result;

        for (const auto& dim : dimensions) {
            std::string key = prefix_ + dim;
            auto* reply = static_cast<redisReply*>(
                redisCommand(ctx_, "ZREVRANK %s %s", key.c_str(), user.c_str())
            );

            if (reply && reply->type == REDIS_REPLY_INTEGER) {
                result[dim] = reply->integer + 1;
            }

            if (reply) freeReplyObject(reply);
        }

        return result;
    }

private:
    redisContext* ctx_;
    std::string prefix_;
};

// 使用
void multiDimensionDemo(redisContext* ctx) {
    MultiDimensionLeaderboard board(ctx, "mygame");

    // 更新用户数据
    board.updateMultiple("player_001", {
        {"score", 95000},
        {"level", 45},
        {"power", 128000},
        {"vip", 8}
    });

    // 查询用户在各榜的排名
    auto ranks = board.getAllRanks("player_001",
        {"score", "level", "power", "vip"});

    for (const auto& [dim, rank] : ranks) {
        std::cout << dim << "榜 排名: " << rank << '\n';
    }
}
```

## 性能优化

### 大数据量下的注意事项

| 操作 | 复杂度 | 百万数据耗时 | 建议 |
|------|-------|-------------|------|
| ZADD | O(log N) | < 1ms | 正常使用 |
| ZREVRANK | O(log N) | < 1ms | 正常使用 |
| ZREVRANGE (Top 100) | O(log N + M) | < 1ms | 正常使用 |
| ZREVRANGE (全量) | O(N) | 很慢 | 禁止 |
| ZCARD | O(1) | < 1ms | 正常使用 |

### 分页查询

```cpp
// 分页获取排行榜
std::vector<RankEntry> getPage(int page, int page_size) {
    int start = (page - 1) * page_size;
    int end = start + page_size - 1;

    return getRange(start + 1, end + 1);
}
```

### 缓存 Top N

对于访问频繁的 Top N，可以加一层本地缓存：

```cpp
class CachedLeaderboard {
public:
    std::vector<RankEntry> getTopN(int n) {
        auto now = std::chrono::steady_clock::now();

        // 缓存有效期内直接返回
        if (!cache_.empty() &&
            now - cache_time_ < std::chrono::seconds(5)) {
            return std::vector<RankEntry>(
                cache_.begin(),
                cache_.begin() + std::min(n, static_cast<int>(cache_.size()))
            );
        }

        // 刷新缓存
        cache_ = board_.getTopN(100);  // 预取 100 条
        cache_time_ = now;

        return std::vector<RankEntry>(
            cache_.begin(),
            cache_.begin() + std::min(n, static_cast<int>(cache_.size()))
        );
    }

private:
    Leaderboard board_;
    std::vector<RankEntry> cache_;
    std::chrono::steady_clock::time_point cache_time_;
};
```

## 最佳实践

| 推荐 | 避免 |
|-----|------|
| 排名从 1 开始展示 | 直接展示 Redis 的 0-based 排名 |
| 使用 ZINCRBY 增量更新 | 每次 ZSCORE 后再 ZADD |
| 周期榜设置过期时间 | 历史数据永不清理 |
| 分数相同时考虑时间因素 | 忽略同分排序问题 |
| 分页查询，限制返回数量 | ZRANGE 0 -1 全量查询 |
